\section{One Layer}

\begin{equation}
\left\lbrace
\begin{array}{rcl}
\partial_t E_{00} & = & D_{01}-A_{01} + D_{02}-A_{02}\\
\partial_t E_{01} & = & -D_{01}+A_{01}\\
\partial_t E_{02} & = & -D_{02}+A_{02}\\
1                 & = & E_{00}+E_{01}+E_{02} = \sum_{x\leq y} E_{xy}\\
\end{array}
\right.
\end{equation}
with
\begin{equation}
\left\lbrace
\begin{array}{rcl}
A_{01} = a_{01} E_{00} C_1\\
A_{02} = a_{02} E_{00} C_2\\
D_{01} = d_{01} E_{01}\\
D_{02} = d_{02} E_{02}\\
\end{array}
\right.
\end{equation}

\begin{equation}
\partial_t
\begin{bmatrix}
E_{00}\\
E_{01}\\
E_{02}\\
\end{bmatrix}
=
\begin{bmatrix}
0 & d_{01} & d_{02}\\
0 & -d_{01} & 0\\
0 & 0 & -d_{02}\\
\end{bmatrix}
\begin{bmatrix}
E_{00}\\
E_{01}\\
E_{02}\\
\end{bmatrix}
+
\begin{bmatrix}
-a_{01}E_{00} & - a_{02} E_{00}\\
a_{01} E_{00} & 0 \\
0 & a_{02} E_{00} \\
\end{bmatrix}
\begin{bmatrix}
C_1\\
C_2\\
\end{bmatrix}
\end{equation}
